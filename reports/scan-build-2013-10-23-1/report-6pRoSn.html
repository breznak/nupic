127// FUNCTION: _smooth()
128// PURPOSE: Smooth a 1D histogram
129float _smoothHist1D(const float * pfHistogram,
130 float * pfReflHist,
131 float * pfSmoothHist,
132 int nHistWidth) {
133
134 int i, k;
135 const float * pfWindowPtr;
136 float fAccum;
137 float fRef = 2.0 * (*pfHistogram);
138
139 // Construct reflected (extended) histogrm
140 float * pfReflHistPtr = pfReflHist + HANNING_HALF_LEN;
141 const float * pfHistPtr = pfHistogram;
142 // Reflect leading elements
143 for (k=HANNING_HALF_LEN; k; k--)
144 *--pfReflHistPtr = fRef - *++pfHistPtr;
145 // Copy internal elemenents
146 pfHistPtr = pfHistogram;
147 pfReflHistPtr += HANNING_HALF_LEN;
148 for (k=nHistWidth; k; k--)
149 *pfReflHistPtr++ = *pfHistPtr++;
150 // Reflect trailing elements
151 fRef = 2.0 * (*--pfHistPtr);
152 for (k=HANNING_HALF_LEN; k; k--)
153 *pfReflHistPtr++ = fRef - *--pfHistPtr;
154
155 // Perform convolution
156 float fMaxVal = -MAXFLOAT;
157 float * pfSmoothHistPtr = pfSmoothHist;
158 for (k=nHistWidth; k; k--) {
159 pfReflHistPtr = pfReflHist;
160 pfWindowPtr = afHanning;
161 fAccum = 0.0f;
162 for (i=HANNING_LEN; i; i--)
163 fAccum += (*pfWindowPtr++) * (*pfReflHistPtr++);
164 *pfSmoothHistPtr++ = fAccum;
165 if (fAccum > fMaxVal)
166 fMaxVal = fAccum;
167 pfReflHist++;
168 }
169
170 return fMaxVal;
171}
