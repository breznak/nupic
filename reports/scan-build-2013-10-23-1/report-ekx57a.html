756int main(int argc, char *argv[])
757{
758
759 /*
760 * Without arguments, this program is a simple end-to-end demo
761 * of NuPIC 2 functionality, used as a developer tool (when
762 * we add a feature, we add it to this program.
763 * With an integer argument N, runs the same test N times
764 * and requires that memory use stay constant -- it can't
765 * grow by even one byte.
766 */
767
768 // TODO: real argument parsing
769 // Optional arg is number of iterations to do.
770 NTA_CHECK(argc == 1 || argc == 2)if (argc == 1 || argc == 2) {} else throw nta::LoggingException
("/home/marek/devel/nupic/nta/apps/htmtest/htmtest.cpp", 770)
<< "CHECK FAILED: \"" << "argc == 1 || argc == 2"
<< "\" ";
771 size_t count = 1;
772 if (argc == 2)
773 {
774 std::stringstream ss(argv[1]);
775 ss >> count;
776 }
777 // Start checking memory usage after this many iterations.
778#ifdef NTA_PLATFORM_win32
779 // takes longer to settle down on win32
780 size_t memoryLeakStartIter = 6000;
781#else
782 size_t memoryLeakStartIter = 150;
783#endif
784
785 // This determines how frequently we check.
786 size_t memoryLeakDeltaIterCheck = 10;
787
788 size_t minCount = memoryLeakStartIter + 5 * memoryLeakDeltaIterCheck;
789
790 if (count > 1 && count < minCount)
791 {
792 std::cout << "Run count of " << count << " specified\n";
793 std::cout << "When run in leak detection mode, count must be at least " << minCount << "\n";
794 ::exit(1);
795 }
796
797
798 size_t initial_vmem = 0;
799 size_t initial_rmem = 0;
800 size_t current_vmem = 0;
801 size_t current_rmem = 0;
802 try {
803 for (size_t i = 0; i < count; i++)
804 {
805 //MemoryMonitor m;
806 NuPIC::init();
807 realmain(count > 1);
808 //testExceptionBug();
809 //testPynode1xLinking();
810 // testNuPIC1x();
811 //testSecondTimeLeak();
812 //testPynodeLinking();
813 //testCppLinking("TestFanIn2","");
814 NuPIC::shutdown();
815 // memory leak detection
816 // we check even prior to the initial tracking iteration, because the act
817 // of checking potentially modifies our memory usage
818 if (i % memoryLeakDeltaIterCheck == 0)
819 {
820 OS::getProcessMemoryUsage(current_rmem, current_vmem);
821 if(i == memoryLeakStartIter)
822 {
823 initial_rmem = current_rmem;
824 initial_vmem = current_vmem;
825 }
826 std::cout << "Memory usage: " << current_vmem << " (virtual) "
827 << current_rmem << " (real) at iteration " << i << std::endl;
828
829 if(i >= memoryLeakStartIter)
830 {
831 if (current_vmem > initial_vmem || current_rmem > initial_rmem)
832 {
833 std::cout << "Tracked memory usage (iteration "
834 << memoryLeakStartIter << "): " << initial_vmem
835 << " (virtual) " << initial_rmem << " (real)" << std::endl;
836 throw std::runtime_error("Memory leak detected");
837 }
838 }
839 }
840 }
841
842 } catch (nta::Exception& e) {
843 std::cout
844 << "Exception: " << e.getMessage()
845 << " at: " << e.getFilename() << ":" << e.getLineNumber()
846 << std::endl;
847 return 1;
848
849 } catch (std::exception& e) {
850 std::cout << "Exception: " << e.what() << "" << std::endl;
851 return 1;
852 }
853 catch (...) {
854 std::cout << "\nhtmtest is exiting because an exception was thrown" << std::endl;
855 return 1;
856 }
857 if (count > 20)
858 std::cout << "Memory leak check passed -- " << count << " iterations" << std::endl;
859
860 std::cout << "--- ALL TESTS PASSED ---" << std::endl;
861 return 0;
862}
