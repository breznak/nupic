117void VectorFileSensor::compute()
118{
119 // It's not necessarily an error to have no outputs. In this case we just return
120 if (dataOut_.getCount() == 0)
121 return;
122
123 // Don't write if there is no open file.
124 if (recentFile_ == "")
125 {
126 NTA_WARNnta::LogItem("/home/marek/devel/nupic/nta/regions/VectorFileSensor.cpp"
, 126, nta::LogItem::warn).stream() << "VectorFileSesnsor compute() called, but there is no open file";
127 return;
128 }
129
130 NTA_CHECK(vectorFile_.vectorCount() > 0)if (vectorFile_.vectorCount() > 0) {} else throw nta::LoggingException
("/home/marek/devel/nupic/nta/regions/VectorFileSensor.cpp", 130
) << "CHECK FAILED: \"" << "vectorFile_.vectorCount() > 0"
<< "\" "
131 << "VectorFileSensor::compute - no data vectors in memory."
132 << "Perhaps no data file has been loaded using the 'loadFile'"
133 << " execute command.";
134
135 if (iterations_ % repeatCount_ == 0) {
136 // Get index to next vector and copy scaled vector to our output
137 curVector_++;
138 curVector_ %= vectorFile_.vectorCount();
139 }
140
141 Real *out = (Real *) dataOut_.getBuffer();
142
143 Size count = dataOut_.getCount();
144 UInt offset = 0;
145
146 if (hasCategoryOut_)
147 {
148 Real * categoryOut = (Real32 *) categoryOut_.getBuffer();
149 vectorFile_.getRawVector((nta::UInt)curVector_, categoryOut, offset, 1);
150 offset++;
151 }
152
153 if (hasResetOut_)
154 {
155 Real * resetOut = (Real32 *)resetOut_.getBuffer();
156 vectorFile_.getRawVector((nta::UInt)curVector_, resetOut, offset, 1);
157 offset++;
158 }
159
160 vectorFile_.getScaledVector((nta::UInt)curVector_, out, offset, count);
161 iterations_++;
162}

175 return 0;
