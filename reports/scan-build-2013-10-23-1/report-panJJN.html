149 template <class T>
150 inline std::size_t float_hash_impl2(T v)
151 {
152 boost::hash_detail::call_frexp<T> frexp;
153 boost::hash_detail::call_ldexp<T> ldexp;
154
155 int exp = 0;
156
157 v = frexp(v, &exp);
158
159 // A postive value is easier to hash, so combine the
160 // sign with the exponent and use the absolute value.
161 if(v < 0) {
162 v = -v;
163 exp += limits<T>::max_exponent -
164 limits<T>::min_exponent;
165 }
166
167 v = ldexp(v, limits<std::size_t>::digits);
168 std::size_t seed = static_cast<std::size_t>(v);
169 v -= static_cast<T>(seed);
170
171 // ceiling(digits(T) * log2(radix(T))/ digits(size_t)) - 1;
172 std::size_t const length
173 = (limits<T>::digits *
174 boost::static_log2<limits<T>::radix>::value
175 + limits<std::size_t>::digits - 1)
176 / limits<std::size_t>::digits;
177
178 for(std::size_t i = 0; i != length; ++i)
179 {
180 v = ldexp(v, limits<std::size_t>::digits);
181 std::size_t part = static_cast<std::size_t>(v);
182 v -= static_cast<T>(part);
183 hash_float_combine(seed, part);
184 }
185
186 hash_float_combine(seed, exp);
187
188 return seed;
