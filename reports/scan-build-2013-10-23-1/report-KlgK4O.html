87//////////////////////////////////////////////////////////////////////////
88/// open the given file by name
89/////////////////////////////////////////////////////////////////////////
90void IFStream::open(const char * filename, ios_base::openmode mode)
91{
92#ifdef WIN32
93 std::wstring pathW = Path::utf8ToUnicode(filename);
94 std::ifstream::open(pathW.c_str(), mode);
95#else
96 std::ifstream::open(filename, mode);
97#endif
98
99 // Check for error
100 if (!is_open()) {
101 IFStream::diagnostics(filename);
102 // On unix, running nfs, we occasionally get errors opening a file on an nfs drive
103 // and it seems that simply doing a retry makes it successful
104 #ifndef WIN32
105 std::ifstream::clear();
106 std::ifstream::open(filename, mode);
107 #endif
108 }
109}
